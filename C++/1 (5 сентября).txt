// Online C++ compiler to run C++ program online
#include <iostream>
#include <math.h>
using namespace std;
int main()
{
    int i;
    float a,b,c,p,s;
    cout<< "\n Код запущен";
    while (true)
    {
        cout<<"\na="; cin>>a;
        cout<<"\nb="; cin>>b;
        cout<<"\nc="; cin>>c;
        p=(a+b+c)/2;
        s=sqrt(p*(p-a)*(p-b)*(p-c));
        if (s > 15)
        {
            cout<<"\n Площадь большая ппц, находите меньше пожалуйста хехе";
        }
        else
        {
            cout<<"\n Площадь маленькая, молодец мурмур ^_^";
        }
        cout << "\n Площадь Треугольника = " << s;
    }
    cout<<"\n Конец кода";
    return 0;
}
